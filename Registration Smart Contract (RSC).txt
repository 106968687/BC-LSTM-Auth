pragma solidity ^0.8.17;

/**
 * @title RegistrationSmartContract (RSC)
 * @dev Implements immutable ID-PK binding and identity registration for data sources
 */
contract RegistrationSmartContract {
    // Mapping: Data source ID -> Public key (bytes256)
    mapping(string => bytes256) public idToPublicKey;
    // Mapping: Public key -> Data source ID (string)
    mapping(bytes256 => string) public publicKeyToId;
    // Mapping: Data source ID -> Registration timestamp (uint256)
    mapping(string => uint256) public registrationTimestamp;
    
    // Event emitted when a data source is successfully registered
    event Registered(string indexed dataSourceId, bytes256 indexed publicKey, uint256 timestamp);
    
    /**
     * @dev Register a new data source with its unique ID and public key
     * @param dataSourceId Unique identifier of the data source (string)
     * @param publicKey 256-bit ECC public key of the data source (bytes256)
     */
    function register(string calldata dataSourceId, bytes256 calldata publicKey) external {
        // Ensure the ID is not already registered
        require(bytes(idToPublicKey[dataSourceId]).length == 0, "Data source ID already registered");
        // Ensure the public key is not already bound to another ID
        require(bytes(publicKeyToId[publicKey]).length == 0, "Public key already bound to another ID");
        
        // Bind ID and public key
        idToPublicKey[dataSourceId] = publicKey;
        publicKeyToId[publicKey] = dataSourceId;
        registrationTimestamp[dataSourceId] = block.timestamp;
        
        // Emit registration event
        emit Registered(dataSourceId, publicKey, block.timestamp);
    }
    
    /**
     * @dev Verify if a data source ID is registered and its public key is valid
     * @param dataSourceId Unique identifier of the data source
     * @param publicKey Public key to verify
     * @return True if the ID-PK binding is valid, false otherwise
     */
    function verifyIdentity(string calldata dataSourceId, bytes256 calldata publicKey) external view returns (bool) {
        return idToPublicKey[dataSourceId] == publicKey;
    }
    
    /**
     * @dev Get the public key bound to a given data source ID
     * @param dataSourceId Unique identifier of the data source
     * @return Public key (bytes256) bound to the ID
     */
    function getPublicKey(string calldata dataSourceId) external view returns (bytes256) {
        return idToPublicKey[dataSourceId];
    }
    
    /**
     * @dev Get the data source ID bound to a given public key
     * @param publicKey Public key of the data source
     * @return Data source ID (string) bound to the public key
     */
    function getDataSourceId(bytes256 calldata publicKey) external view returns (string memory) {
        return publicKeyToId[publicKey];
    }
}
