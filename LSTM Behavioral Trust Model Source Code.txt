python
import numpy as np
import pandas as pd
from sklearn.preprocessing import MinMaxScaler
from sklearn.model_selection import train_test_split
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense, Dropout
from tensorflow.keras.optimizers import Adam

class LSTMBehavioralTrustModel:
    def __init__(self, sequence_length=10, n_features=4, units=64, dropout_rate=0.3, learning_rate=1e-3):
        """
        Initialize the LSTM Behavioral Trust Model
        :param sequence_length: Length of input time sequences (number of historical behavioral traces)
        :param n_features: Number of input features (timestamp, IP location hash, data volume, signature frequency)
        :param units: Number of units in each LSTM layer
        :param dropout_rate: Dropout rate for regularization
        :param learning_rate: Learning rate for Adam optimizer
        """
        self.sequence_length = sequence_length
        self.n_features = n_features
        self.scaler = MinMaxScaler(feature_range=(0, 1))
        self.model = self._build_model(units, dropout_rate, learning_rate)
    
    def _build_model(self, units, dropout_rate, learning_rate):
        """Build and compile the LSTM model"""
        model = Sequential()
        # First LSTM layer with return_sequences=True (output sequence for next LSTM layer)
        model.add(LSTM(units=units, return_sequences=True, input_shape=(self.sequence_length, self.n_features)))
        model.add(Dropout(dropout_rate))
        # Second LSTM layer
        model.add(LSTM(units=units, return_sequences=False))
        model.add(Dropout(dropout_rate))
        # Dense output layer (sigmoid activation for binary classification: normal/anomalous)
        model.add(Dense(units=1, activation='sigmoid'))
        
        # Compile the model
        optimizer = Adam(learning_rate=learning_rate)
        model.compile(optimizer=optimizer, loss='binary_crossentropy', metrics=['accuracy', 'Precision', 'Recall'])
        return model
    
    def preprocess_data(self, data):
        """
        Preprocess the input behavioral data: normalize and create sequences
        :param data: Input data (pandas DataFrame) with 4 features: timestamp, loc_hash, volume, frequency
        :return: X (sequences), y (labels: 0=normal, 1=anomalous)
        """
        # Extract features and labels
        features = data[['timestamp', 'loc_hash', 'volume', 'frequency']].values
        labels = data['label'].values  # label: 0=normal, 1=anomalous
        
        # Normalize features
        features_scaled = self.scaler.fit_transform(features)
        
        # Create sequences (input: sequence_length consecutive samples, output: label of the last sample)
        X, y = [], []
        for i in range(self.sequence_length, len(features_scaled)):
            X.append(features_scaled[i-self.sequence_length:i, :])
            y.append(labels[i])
        
        # Convert to numpy arrays
        X = np.array(X)
        y = np.array(y)
        return X, y
    
    def train(self, train_data, epochs=50, batch_size=32, validation_split=0.2):
        """
        Train the LSTM model
        :param train_data: Training data (pandas DataFrame) with features and labels
        :param epochs: Number of training epochs
        :param batch_size: Batch size for training
        :param validation_split: Fraction of training data used for validation
        :return: Training history
        """
        X_train, y_train = self.preprocess_data(train_data)
        history = self.model.fit(
            X_train, y_train,
            epochs=epochs,
            batch_size=batch_size,
            validation_split=validation_split,
            shuffle=False  # Shuffle is False for time series data
        )
        return history
    
    def predict(self, test_data):
        """
        Predict anomaly scores for test data
        :param test_data: Test data (pandas DataFrame) with features and labels
        :return: Predictions (0=normal, 1=anomalous) and true labels
        """
        X_test, y_test = self.preprocess_data(test_data)
        predictions = (self.model.predict(X_test) > 0.5).astype(int)  # Threshold 0.5 for binary classification
        return predictions, y_test
    
    def save_model(self, path):
        """Save the trained model to a file"""
        self.model.save(path)
    
    def load_model(self, path):
        """Load a trained model from a file"""
        from tensorflow.keras.models import load_model
        self.model = load_model(path)

# Example usage
